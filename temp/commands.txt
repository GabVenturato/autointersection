- Basic instructions

1) RUN in separate shells:

erl -sname v1 -pa ebin/
erl -sname v2 -pa ebin/
erl -sname v3 -pa ebin/
erl -sname env -pa ebin/

2) Execute in env.

environment:start().
gen_server:call(environment, {occupy_position, spawn(fun() -> timer:sleep(60000),
                 gen_server:call(environment, {release_position, "R_5"}) end), "R_5"}).

3) QUICKLY execute in v3, v2, v1 respecting the order (timer in previous step set to 60s):
	Make sure to modify the tuple {Name, Node} before !!


Route = gen_server:call({environment, env@lenora}, {get_route, {"R_4", "R_30"}}),
vehicle:start_vehicle(Route, {environment, env@lenora}),
gen_server:cast({coordinator, v3@lenora}, {startup}).

Route = gen_server:call({environment, env@lenora}, {get_route, {"R_2", "R_30"}}),
vehicle:start_vehicle(Route, {environment, env@lenora}),
gen_server:cast({coordinator, v2@lenora}, {startup}).

Route = gen_server:call({environment, env@lenora}, {get_route, {"R_1", "R_30"}}),
vehicle:start_vehicle(Route, {environment, env@lenora}),
gen_server:cast({coordinator, v1@lenora}, {startup}).


4) Alternative: to simulate mechanical failure run after 3) in node v3:
